let settings = {};
let activeTab = 'models';

export async function render(container) {
  try {
    settings = (await window.openClaw.settings.get()) || {};
  } catch(e) {
    settings = {};
  }
  if (!settings.providers) settings.providers = {};
  
  // 处理配置状态
  const configured = [];
  const unconfigured = [];
  
  allProviders.forEach(p => {
    const pConf = settings.providers[p.id];
    if (pConf && pConf.apiKey && pConf.apiKey.trim() !== '') {
      configured.push({ ...p, conf: pConf });
    } else {
      unconfigured.push({ ...p, conf: pConf || {} });
    }
  });

  // 处理自定义模型 (如果 settings 里有我们不在预设列表的)
  Object.keys(settings.providers).forEach(key => {
    if (key.startsWith('custom_') && settings.providers[key].apiKey) {
      configured.push({
         id: key, 
         name: settings.providers[key].customName || '自定义模型', 
         icon: settings.providers[key].customLogo || '🔧',
         url: settings.providers[key].baseUrl,
         conf: settings.providers[key],
         isCustom: true
      });
    }
  });

  container.className = 'page-layout-full';
  
  container.innerHTML = `
    <div class="page-header" style="display: flex; gap: 24px; border-bottom: 1px solid var(--border-light); padding-bottom: 0;">
      <h2 style="margin: 0; padding-bottom: 16px; font-size: 24px; font-weight: 600;">⚙️ 设置</h2>
      <div style="display: flex; gap: 16px;">
         <div class="tab-btn ${activeTab === 'models' ? 'active' : ''}" data-tab="models" style="padding-bottom: 16px; cursor: pointer; border-bottom: 2px solid ${activeTab === 'models' ? 'var(--primary)' : 'transparent'}; color: ${activeTab === 'models' ? 'var(--text-primary)' : 'var(--text-muted)'}; font-weight: 500;">模型配置中心</div>
         <div class="tab-btn ${activeTab === 'general' ? 'active' : ''}" data-tab="general" style="padding-bottom: 16px; cursor: pointer; border-bottom: 2px solid ${activeTab === 'general' ? 'var(--primary)' : 'transparent'}; color: ${activeTab === 'general' ? 'var(--text-primary)' : 'var(--text-muted)'}; font-weight: 500;">通用偏好</div>
      </div>
    </div>
    
    <div class="page-content" id="settingsContent" style="padding-top: 24px;">
       <!-- 动态内容渲染区 -->
    </div>
  `;

  document.querySelectorAll('.tab-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      activeTab = btn.dataset.tab;
      render(container);
    });
  });

  renderContent(configured, unconfigured);
}

                    <div>
                       <div style="font-weight: 600; font-size: 15px;">${p.name}</div>
                       <div style="font-size: 12px; color: var(--text-muted); margin-top: 4px; font-family: monospace;">${p.url || p.conf.baseUrl}</div>
                    </div>

           <div class="card" style="padding: 20px; background: rgba(0, 122, 255, 0.05); border: 1px dashed var(--primary);">
              <div style="display: grid; grid-template-columns: 1fr 1fr; gap: 16px; margin-bottom: 16px;">
                 <div>
                   <label style="display: block; font-size: 13px; font-weight: 500; margin-bottom: 6px; color: var(--text-muted);">模型名称</label>
                   <input type="text" id="customName" class="input" placeholder="如: Ollama Llama3 或 My Custom API" style="width: 100%;" />
                 </div>
                 <div>
                   <label style="display: block; font-size: 13px; font-weight: 500; margin-bottom: 6px; color: var(--text-muted);">自定义 Logo / Emoji</label>
                   <input type="text" id="customLogo" class="input" placeholder="🔧" value="🔧" style="width: 100%;" />
                 </div>
                 <div style="grid-column: 1 / -1;">
                   <label style="display: block; font-size: 13px; font-weight: 500; margin-bottom: 6px; color: var(--text-muted);">API 请求地址 (URL)</label>
                   <input type="text" id="customUrl" class="input" placeholder="如: http://127.0.0.1:11434/v1" style="width: 100%;" />
                 </div>
                 <div style="grid-column: 1 / -1;">
                   <label style="display: block; font-size: 13px; font-weight: 500; margin-bottom: 6px; color: var(--text-muted);">鉴权密钥 (API Key)</label>
                   <input type="password" id="customKey" class="input" placeholder="sk-..." style="width: 100%;" />
                 </div>
              </div>
              <button class="btn btn-primary" id="addCustomBtn">保存自定义模型</button>
           </div>
        </div>

        <div>
           <h3 style="font-size: 16px; font-weight: 600; margin: 0 0 16px 0; display: flex; align-items: center; gap: 8px;">

             `).join('')}
           </div>
        </div>
      </div>
    `;

    document.getElementById('addCustomBtn').addEventListener('click', async () => {
      const name = document.getElementById('customName').value.trim();
      const logo = document.getElementById('customLogo').value.trim();
      const url = document.getElementById('customUrl').value.trim();
      const key = document.getElementById('customKey').value.trim();
      if (!name || !url) {
        if (window.__toast) window.__toast.error('模型名称和URL不能为空');
        return;
      }
      const id = 'custom_' + Date.now();
      settings.providers[id] = { baseUrl: url, apiKey: key, customName: name, customLogo: logo };
      await window.openClaw.settings.save(settings);
      if (window.__toast) window.__toast.success('自定义模型添加成功！');
      render(document.getElementById('pageContainer'));
    });

    document.querySelectorAll('.edit-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const id = btn.dataset.id;
        const pConf = settings.providers[id] || {};
        const isCustom = id.startsWith('custom_');
        const defaultUrl = isCustom ? pConf.baseUrl : (allProviders.find(x => x.id === id)?.url || '');
        const newKey = prompt(`请输入 [${id}] 的 API Key:`, pConf.apiKey || '');
        if (newKey !== null) {
          let newUrl = prompt(`请输入 [${id}] 的 API URL:`, pConf.baseUrl || defaultUrl);
          if (newUrl !== null) {
            settings.providers[id] = { ...pConf, apiKey: newKey, baseUrl: newUrl };
            window.openClaw.settings.save(settings).then(() => {
              if (window.__toast) window.__toast.success('配置已更新');
              render(document.getElementById('pageContainer'));
            });
          }
        }
      });
    });

    document.querySelectorAll('.delete-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const id = btn.dataset.id;
        if (confirm(`确定要删除/清空 [${id}] 的配置吗？`)) {
          settings.providers[id] = { apiKey: '', baseUrl: '' };
          window.openClaw.settings.save(settings).then(() => {
            if (window.__toast) window.__toast.info('配置已清除');
            render(document.getElementById('pageContainer'));
          });
        }
      });
    });

        </div>

    `;
