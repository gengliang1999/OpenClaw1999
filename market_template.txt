 * 提供 厂商 -> 系列/模型 的三级分类浏览
 */

let currentLevel = 'vendor'; // vendor, models
let currentVendor = null;

const vendors = [
  { id: 'ollama', name: 'Ollama 本地模型', icon: '🦙', isLocal: true, desc: '运行在本地的轻量级开源大模型' },
  { id: 'lmstudio', name: 'LM Studio 镜像源', icon: '💻', isLocal: true, desc: '支持从 ModelScope / HuggingFace 国内镜像下载 GGUF 模型' },
  { id: 'openai', name: 'OpenAI', icon: '🌌', isLocal: false, desc: 'GPT-4o, GPT-3.5-Turbo 等旗舰模型' },
  { id: 'anthropic', name: 'Anthropic', icon: '🧠', isLocal: false, desc: 'Claude 3.5 Sonnet 等强力逻辑模型' },
  { id: 'deepseek', name: 'DeepSeek', icon: '🐳', isLocal: false, desc: '国产深度求索，高性价比代码模型' },
  { id: 'qwen', name: '通义千问', icon: '☁️', isLocal: false, desc: '阿里云通义千问全系列' },
  { id: 'zhipu', name: '智谱 AI', icon: '🔮', isLocal: false, desc: 'GLM-4, GLM-4V 多模态系列' }
];

const mockModels = {
  ollama: [
    { id: 'llama3', name: 'Llama 3 8B', size: '4.7GB', tags: ['指令跟随', '快速'] },
    { id: 'qwen2-7b', name: 'Qwen 2 7B', size: '4.4GB', tags: ['中文优化', '全能'] }
  ],
  lmstudio: [
    { id: 'glm4-9b', name: 'GLM 4 9B (GGUF)', size: '5.2GB', tags: ['魔塔镜像', '量化'] }
  ],
  openai: [
    { id: 'gpt-4o', name: 'GPT-4o', size: 'Cloud', tags: ['旗舰', '多模态'] },
    { id: 'gpt-4-turbo', name: 'GPT-4 Turbo', size: 'Cloud', tags: ['逻辑能力'] }
  ],
  anthropic: [
    { id: 'claude-3-5-sonnet', name: 'Claude 3.5 Sonnet', size: 'Cloud', tags: ['高速', '代码生成'] }
  ]
};

export async function render(container) {
  container.className = 'page-layout-full';
  
  container.innerHTML = `
    <div class="page-header" style="display: flex; flex-direction: column; gap: 16px;">
      <div style="display: flex; justify-content: space-between; align-items: center;">
        <h2 style="margin: 0; font-size: 24px; font-weight: 600;">🛒 模型市场</h2>
        <div style="font-size: 13px; color: var(--success); display: flex; align-items: center; gap: 6px; background: rgba(52, 199, 89, 0.1); padding: 6px 12px; border-radius: 20px;">
           <span style="width: 8px; height: 8px; background: var(--success); border-radius: 50%; display: inline-block;"></span>
           国内镜像源已连通 (ModelScope 同步)
        </div>
      </div>
      
      <!-- 面包屑导航 -->
      <div id="breadcrumb" class="breadcrumb" style="display: flex; align-items: center; gap: 8px; font-size: 14px; font-weight: 500;">
         <span class="crumb-link" data-target="vendor" style="color: var(--primary); cursor: pointer; display: flex; align-items: center; gap: 4px;">🏠 厂商库</span>
      </div>
    </div>
    
    <div class="page-content" id="marketContent" style="padding-top: 0;">
       <!-- 动态内容渲染区 -->
    </div>
  `;

  document.getElementById('breadcrumb').addEventListener('click', (e) => {
    const link = e.target.closest('.crumb-link');
    if (!link) return;
    const target = link.dataset.target;
    if (target === 'vendor') {
      currentLevel = 'vendor';
      currentVendor = null;
      renderView();
    }
  });

  renderView();
}

function renderView() {
  const content = document.getElementById('marketContent');
  const breadcrumb = document.getElementById('breadcrumb');
  
  if (currentLevel === 'vendor') {
    breadcrumb.innerHTML = `<span class="crumb-link" data-target="vendor" style="color: var(--text-primary); cursor: default;">🏠 厂商库</span>`;
    
    content.innerHTML = `
      <div class="grid" style="display: grid; grid-template-columns: repeat(auto-fill, minmax(280px, 1fr)); gap: 20px; margin-top: 20px;">
        ${vendors.map(v => `
          <div class="card vendor-card" data-id="${v.id}" style="cursor: pointer; position: relative;">
            <div style="position: absolute; top: 16px; right: 16px; font-size: 12px; font-weight: 600; padding: 4px 8px; border-radius: 12px; background: ${v.isLocal ? 'rgba(0, 217, 255, 0.1)' : 'rgba(108, 99, 255, 0.1)'}; color: ${v.isLocal ? '#00d9ff' : '#6c63ff'}; border: 1px solid currentColor;">
               ${v.isLocal ? '本地' : '云端'}
            </div>
            <div style="font-size: 40px; margin-bottom: 16px;">${v.icon}</div>
            <h3 style="margin: 0 0 8px 0; font-size: 18px;">${v.name}</h3>
            <p style="margin: 0; font-size: 13px; color: var(--text-muted); line-height: 1.5;">${v.desc}</p>
          </div>
        `).join('')}
      </div>
    `;

    document.querySelectorAll('.vendor-card').forEach(card => {
      card.addEventListener('click', () => {
        const id = card.dataset.id;
        currentVendor = vendors.find(x => x.id === id);
        currentLevel = 'models';
        renderView();
      });
    });
  } else if (currentLevel === 'models') {
    breadcrumb.innerHTML = `
      <span class="crumb-link" data-target="vendor" style="color: var(--primary); cursor: pointer;">🏠 厂商库</span>
      <span style="color: var(--text-muted);">/</span>
      <span style="color: var(--text-primary);">${currentVendor.name} 模型</span>
    `;

    const modelsList = mockModels[currentVendor.id] || [];
    
    if (modelsList.length === 0) {
      content.innerHTML = `
        <div style="text-align: center; padding: 60px 20px; color: var(--text-muted);">
           <div style="font-size: 48px; margin-bottom: 16px;">🚧</div>
           <div>此厂商暂无可用模型，或需刷新镜像源缓存。</div>
        </div>
      `;
      return;
    }

    content.innerHTML = `
      <div style="margin-top: 20px; display: flex; flex-direction: column; gap: 12px;">
        ${modelsList.map(m => `
          <div class="card" style="display: flex; align-items: center; justify-content: space-between; padding: 16px 20px;">
             <div style="display: flex; align-items: center; gap: 16px;">
                <div style="width: 40px; height: 40px; background: var(--bg-active); border-radius: 8px; display: flex; align-items: center; justify-content: center; font-size: 20px;">
                   ${currentVendor.icon}
                </div>
                <div>
                   <h4 style="margin: 0 0 4px 0; font-size: 16px;">${m.name}</h4>
                   <div style="display: flex; gap: 8px;">
                     ${m.tags.map(t => `<span style="font-size: 11px; background: var(--bg-app); padding: 2px 6px; border-radius: 4px; border: 1px solid var(--border-light);">${t}</span>`).join('')}
                     <span style="font-size: 11px; color: var(--text-muted); padding: 2px 6px;">Size: ${m.size}</span>
                   </div>
                </div>
             </div>
             <button class="btn btn-primary btn-dl" style="padding: 6px 16px; border-radius: 20px;">
               ${currentVendor.isLocal ? '下载部署' : '前往配置'}
             </button>
          </div>
        `).join('')}
      </div>
    `;

    document.querySelectorAll('.btn-dl').forEach(btn => {
       btn.addEventListener('click', () => {
          if (currentVendor.isLocal) {
             if (window.__toast) window.__toast.success('已加入国内镜像下载队列！');
          } else {
             window.location.hash = '#/settings';
             if (window.__toast) window.__toast.info('请在设置中配置对应的 API Key。');
          }
       });
    });
  }
}
